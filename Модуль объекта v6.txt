Функция ОпределитьВремяРег(ТекВремя) Экспорт
	Секунд = Число(ТекущаяДата() - ТекВремя);
	Сек = 0;
	Мин = 0;
	Час = 0;
	Если Цел(Секунд / 3600) <> 0 Тогда
		Час = Цел(Секунд / 3600);
	КонецЕсли;

	Ост = Секунд - Час * 3600;
	Если Цел(Ост / 60) <> 0 Тогда
		Мин = Цел(Ост / 60);
	КонецЕсли;

	Сек = Секунд - Час * 3600 - Мин * 60;
	время = ?(Час = 0, "", Строка(Час) + " часов ") + ?(Мин = 0, "", Строка(Мин) + " мин. ") + Строка(Сек) + " сек.";
	Возврат Строка(время);
