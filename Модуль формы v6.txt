&НаКлиенте
Процедура ДеревоОбъектовПометкаПриИзменении(Элемент)
	ИмяОбъекта=Элементы.ДеревоОбъектов.ТекущиеДанные.ИмяОбъекта;	
	ПолноеИмя=Элементы.ДеревоОбъектов.ТекущиеДанные.Тип+"."+ИмяОбъекта;
	Имя=Элементы.ДеревоОбъектов.ТекущиеДанные.Имя;
	Если Элементы.ДеревоОбъектов.ТекущиеДанные.Пометка Тогда
		Если Элементы.ДеревоОбъектов.ТекущиеДанные.получитьРодителя()=Неопределено Тогда
			Кол=1;
			Для каждого Стр Из Элементы.ДеревоОбъектов.ТекущиеДанные.ПолучитьЭлементы() Цикл
				Прогресс=100/Элементы.ДеревоОбъектов.ТекущиеДанные.ПолучитьЭлементы().Количество()*Кол;
				Кол=Кол+1;
				ИмяОбъекта=Стр.ИмяОбъекта;
				Имя=Стр.Имя;
				ПолноеИмя=Стр.Тип+"."+ИмяОбъекта;
				НастройкиСКД=СоздатьСКДНаСервере(ПолноеИмя);
				СтруктураСКД.Вставить(Имя,НастройкиСКД);
				ОбработкаПрерыванияПользователя();
				Состояние("Подождите немного, идет создание схем СКД...",Прогресс);
			КонецЦикла;
			ОчиститьКомпоновщик();
		Иначе
			НастройкиСКД=СоздатьСКДНаСервере(ПолноеИмя);
			СтруктураСКД.Вставить(Имя,НастройкиСКД);
		КонецЕсли;
	Иначе
		Кол=1;
		Если Элементы.ДеревоОбъектов.ТекущиеДанные.получитьРодителя()=Неопределено Тогда
			Для каждого Стр Из Элементы.ДеревоОбъектов.ТекущиеДанные.ПолучитьЭлементы() Цикл
				Прогресс=100/Элементы.ДеревоОбъектов.ТекущиеДанные.ПолучитьЭлементы().Количество()*Кол;
				Кол=Кол+1;
				Если Стр.Пометка Тогда
					Имя=Стр.Имя;
					ОчиститьКомпоновщик();
					СтруктураСКД.Удалить(Имя);
				КонецЕсли;
				ОбработкаПрерыванияПользователя();
				Состояние("Подождите немного,идет создание схем СКД...",Прогресс);
			КонецЦикла;
		Иначе
			ОчиститьКомпоновщик();
			СтруктураСКД.Удалить(Имя);
			Элементы.ДеревоОбъектов.ТекущиеДанные.Отбор=Ложь;
		КонецЕсли;
	КонецЕсли;
КонецПроцедуры

&НаСервере
Функция  СоздатьСКДНаСервере(ПолноеИмя,ВернутьСКД=ЛОЖЬ)
	Об=РеквизитФормыВЗначение("Объект");
	Текст=Об.СформироватьТекстЗапроса(ПолноеИмя);
	СКД=Об.СоздатьСКД(Текст, ПолноеИмя, УникальныйИдентификатор);
	ЗначениеВРеквизитФормы(Об,"Объект");
	Если НЕ ВернутьСКД Тогда
		Возврат СКД.НастройкиПоУмолчанию;
	Иначе
		Возврат СКД;
	КонецЕсли;

&НаСервере
Процедура ОчиститьКомпоновщик()
	СКД=Новый СхемаКомпоновкиДанных;
	АдресСхемы = ПоместитьВоВременноеХранилище(СКД,Новый УникальныйИдентификатор());
	Объект.Компоновщик.Инициализировать(Новый ИсточникДоступныхНастроекКомпоновкиДанных(АдресСхемы));
	Объект.Компоновщик.Настройки.Отбор.Элементы.Очистить();

&НаКлиенте
Процедура ДеревоОбъектовПриАктивизацииСтроки(Элемент)
	Если Элементы.ДеревоОбъектов.ТекущиеДанные.ПолучитьРодителя() <> Неопределено Тогда
		ИмяОбъекта = Элементы.ДеревоОбъектов.ТекущиеДанные.ИмяОбъекта;	
		Если ПолноеИмя = Элементы.ДеревоОбъектов.ТекущиеДанные.Тип + "." + ИмяОбъекта Тогда
			Возврат;
			
		КонецЕсли;	
		
		ПолноеИмя = Элементы.ДеревоОбъектов.ТекущиеДанные.Тип + "." + ИмяОбъекта;
		Имя = Элементы.ДеревоОбъектов.ТекущиеДанные.Имя;
		Если Элементы.ДеревоОбъектов.ТекущиеДанные.Пометка Тогда
			Если ЗначениеЗаполнено(ИмяОбъекта) И СтруктураСКД.Свойство(Имя) Тогда
				ПодключитьОбработчикОжидания("ИнициализироватьСКД", 0.1, Истина);
				
			КонецЕсли;
		Иначе
			ОчиститьКомпоновщик();	
			
		КонецЕсли;
	Иначе
		ОчиститьКомпоновщик();
		
	КонецЕсли;

&НаСервере
Процедура ИнициализироватьСКДНаСервере(ПолноеИмя,Настройки)
	Об=РеквизитФормыВЗначение("Объект");
	Текст=Об.СформироватьТекстЗапроса(ПолноеИмя);
	СКД = Об.СоздатьСКД(Текст, ПолноеИмя, УникальныйИдентификатор);
	Адрес=ПоместитьВоВременноеХранилище(СКД);
	Объект.Компоновщик.ЗагрузитьНастройки(Настройки);
	Объект.Компоновщик.Инициализировать(Новый ИсточникДоступныхНастроекКомпоновкиДанных(Адрес));
	Объект.Компоновщик.Восстановить(СпособВосстановленияНастроекКомпоновкиДанных.ПроверятьДоступность);
	ЗначениеВРеквизитФормы(Об,"Объект");

	Для каждого ЭлементОтбора Из Объект.Компоновщик.Настройки.Отбор.Элементы Цикл
		Если НЕ ЗначениеЗаполнено(Строка(ЭлементОтбора.ЛевоеЗначение)) Тогда
			возврат;
		КонецЕсли;
	КонецЦикла; 
	
	Если Элементы.ДеревоОбъектов.ТекущиеДанные.получитьРодителя()<>Неопределено Тогда
		ЗапомнитьНастройкиВСКД();
	КонецЕсли;
	
	Если Объект.Компоновщик.Настройки.Отбор.Элементы.Количество()>0 Тогда
		Элементы.ДеревоОбъектов.ТекущиеДанные.Отбор=Истина;
	Иначе
		Элементы.ДеревоОбъектов.ТекущиеДанные.Отбор=Ложь;
	КонецЕсли;
КонецПроцедуры

&НаКлиенте
Процедура ЗапомнитьНастройкиВСКД()
	Имя=Элементы.ДеревоОбъектов.ТекущиеДанные.Имя;
	ОтборСКД=СтруктураСКД[Имя].Отбор;
	ОтборСКД.Элементы.Очистить();
	Для каждого ЭлементОтбора Из Объект.Компоновщик.Настройки.Отбор.Элементы Цикл
		ЭлементОтбораСКД=ОтборСКД.Элементы.Добавить(Тип("ЭлементОтбораКомпоновкиДанных"));
		ЭлементОтбораСКД.ПравоеЗначение=ЭлементОтбора.ПравоеЗначение;
		ЭлементОтбораСКД.Видсравнения=ЭлементОтбора.ВидСравнения;
		ЭлементОтбораСКД.ЛевоеЗначение=ЭлементОтбора.ЛевоеЗначение;
		ЭлементОтбораСКД.Использование=ЭлементОтбора.Использование;
	КонецЦикла; 	
КонецПроцедуры

&НаКлиенте
Процедура КомпоновщикПослеУдаления(Элемент)
		Если Объект.Компоновщик.Настройки.Отбор.Элементы.Количество()>0 Тогда

	Иначе
		Элементы.ДеревоОбъектов.ТекущиеДанные.Отбор=Ложь;
		Если Элементы.ДеревоОбъектов.ТекущиеДанные.ПолучитьРодителя()<>Неопределено Тогда
			Имя=Элементы.ДеревоОбъектов.ТекущиеДанные.Имя;
			ОтборСКД=СтруктураСКД[Имя].Отбор;
			ОтборСКД.Элементы.Очистить();
		КонецЕсли;
	КонецЕсли;

&НаСервере
Процедура ПоказатьОбъектыНаСервере(ПолноеИмя, НастройкиСКД)
	
	Результат.Очистить();
	
	КомпоновщикНастроек = Новый КомпоновщикНастроекКомпоновкиДанных;
	КомпоновщикНастроек.ЗагрузитьНастройки(НастройкиСКД);
	
	// Период уже встроен в текст запроса через СформироватьТекстЗапроса().
	// Здесь оставляем только пользовательские отборы из НастройкиСКД.
	КомпоновщикНастроек.Настройки.Выбор.Элементы.Очистить();
	
	ВыбранныеПоля = КомпоновщикНастроек.Настройки.Выбор.Элементы.Добавить(Тип("ВыбранноеПолеКомпоновкиДанных"));
	ВыбранныеПоля.Заголовок = "Ссылка";
	ВыбранныеПоля.Поле = Новый ПолеКомпоновкиДанных("Ссылка");
	ВыбранныеПоля.Использование = Истина;
	
	ДанныеРасшифровки = Новый ДанныеРасшифровкиКомпоновкиДанных;
	КомпоновщикМакета = Новый КомпоновщикМакетаКомпоновкиДанных;
	СКД = СоздатьСКДНаСервере(ПолноеИмя, Истина);
	МакетКомпоновки = КомпоновщикМакета.Выполнить(СКД, КомпоновщикНастроек.Настройки, ДанныеРасшифровки);
	
	ПроцессорКомпоновки = Новый ПроцессорКомпоновкиДанных;
	ПроцессорКомпоновки.Инициализировать(МакетКомпоновки,, ДанныеРасшифровки);
	
	ПроцессорВывода = Новый ПроцессорВыводаРезультатаКомпоновкиДанныхВТабличныйДокумент;
	ПроцессорВывода.УстановитьДокумент(Результат);
	ПроцессорВывода.Вывести(ПроцессорКомпоновки);
	
	АдресДанныхРасшифровки = ПоместитьВоВременноеХранилище(ДанныеРасшифровки, ЭтаФорма.УникальныйИдентификатор);

	ДоступныеДействия.Добавить(ДействиеОбработкиРасшифровкиКомпоновкиДанных.ОткрытьЗначение);
	СтандартнаяОбработка=ложь;

	ПолноеИмя=Элементы.ДеревоОбъектов.ТекущиеДанные.Тип+"."+ИмяОбъекта;

	Если ВыбранноеДействие=ДействиеОбработкиРасшифровкиКомпоновкиДанных.ОткрытьЗначение Тогда
		ОткрытьЗначение(Параметр);
	КонецЕсли;

&НаКлиенте
Процедура Выгрузка(Команда)
	Объект.ВсегоВыгруженоОбъектов=0;
	Объект.ВыгруженныеОбъекты.Очистить();
	Сообщение = Новый СообщениеПользователю;
	Если Объект.ПоказатьКоличествоВыгружаемых Тогда
		Если Вопрос("Показать количество объектов для выгрузки с учетом отборов?
			|Для начала выгрузки нажмите ""Нет""",РежимДиалогаВопрос.ДаНет)=КодВозвратаДиалога.Да Тогда
			Сообщение.Текст=Строка(ТекущаяДата())+" Начало анализа...";
			Сообщение.Сообщить();
			АнализироватьНаСервере(СтруктураСКД);
			Возврат;
		КонецЕсли;
	КонецЕсли;
	
	Если НЕ ЗначениеЗаполнено(Объект.Путь) И НЕ ЗначениеЗаполнено(Объект.Сервер) И Не ЗначениеЗаполнено(Объект.ИмяБазы) Тогда
		Предупреждение("Не указан путь к файлу/базе!");
		Возврат;
	КонецЕсли;
	
	Сообщение.Текст = Строка(ТекущаяДата())+" Начало выгрузки...";
	Сообщение.Сообщить();
	Если Объект.ЧерезCom Тогда
		ВыгрузитьОбъекты(СтруктураСКД);
	Иначе
		Адрес = ВыгрузитьОбъекты(СтруктураСКД);
		ПолучитьФайл(Адрес, Объект.Путь, Ложь);
		
	КонецЕсли;
КонецПроцедуры

&НаСервере
Функция ВыгрузитьОбъекты(СтруктураСКД)
	Колич=0;
	ИмяФайла = КаталогВременныхФайлов()+"temp.xml";
	ТекВремя=ТекущаяДата();
	ЗаписьXML = Новый ЗаписьXML;
	Если Объект.ЧерезCom Тогда
		ЗаписьXML.УстановитьСтроку();
	Иначе
		
		ЗаписьXML.ОткрытьФайл(ИмяФайла);
		ЗаписьXML.ЗаписатьОбъявлениеXML();
		ЗаписьXML.ЗаписатьНачалоЭлемента("Root");
	КонецЕсли;
	СтрокиДерева=ДеревоОбъектов.ПолучитьЭлементы();
	Об=РеквизитФормыВЗначение("Объект");
	Для каждого СтрокаДерева Из СтрокиДерева Цикл
		Для каждого Стр Из СтрокаДерева.ПолучитьЭлементы() Цикл
			Если Стр.Пометка Тогда
				НастройкиСКД=СтруктураСКД[Стр.Имя];
				ПолноеИмя=Стр.Тип+"."+Стр.ИмяОбъекта;
				СКД=СоздатьСКДНаСервере(ПолноеИмя,Истина);
				Колич=Колич+Об.ВыгрузитьОбъектыЧерезСКД(ЗаписьXML,СКД,НастройкиСКД);
			КонецЕсли;
		КонецЦикла;
	КонецЦикла;
	Если НЕ Объект.ЧерезCom Тогда
		ЗаписьXML.ЗаписатьКонецЭлемента();
	КонецЕсли;
	ЗначениеВРеквизитФормы(Об,"Объект");
	Попытка 
		ЗаписьXML.Закрыть();
		Сообщение = Новый СообщениеПользователю;
		Сообщение.Текст = Строка(ТекущаяДата())+" Выгрузка завершена успешно! Выгр. всего объектов: "+Строка(Объект.ВсегоВыгруженоОбъектов)+" Время выгрузки "+Об.ОпределитьВремяРег(ТекВремя);
		Сообщение.Сообщить(); 
	Исключение 
		Сообщение = Новый СообщениеПользователю;
		Сообщение.Текст = ИнформацияОбОшибке().Описание;
		Сообщение.Сообщить(); 
	КонецПопытки;
	Если Не Объект.ЧерезCom Тогда
		Возврат ПоместитьВоВременноеХранилище(Новый ДвоичныеДанные(ИмяФайла),ЭтаФорма.УникальныйИдентификатор);
		//Возврат ИмяФайла;
	Иначе
		Возврат Истина;
	КонецЕсли;
КонецФункции
